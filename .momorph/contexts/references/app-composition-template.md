# App Composition Template

Use this template when bootstrapping Minimal Starter project wiring.

These files are generated by `scripts/scaffold_project.sh` and should be customized, not rewritten from scratch.

## App Container

```swift
import Foundation

@MainActor
final class AppContainer {
    let appStore: AppStore
    let loadHomeUseCase: LoadHomeUseCase

    init(
        appStore: AppStore = .init(),
        loadHomeUseCase: LoadHomeUseCase
    ) {
        self.appStore = appStore
        self.loadHomeUseCase = loadHomeUseCase
    }
}
```

## App Route

```swift
import Foundation

enum AppRoute: Hashable, Sendable {
    case home
    case settings
}
```

## App Entry

```swift
import SwiftUI

@main
struct MyApp: App {
    @State private var container = AppContainer(
        loadHomeUseCase: PreviewLoadHomeUseCase()
    )

    var body: some Scene {
        WindowGroup {
            RootView(container: container)
        }
    }
}
```

## Root View

```swift
import SwiftUI

struct RootView: View {
    let container: AppContainer
    @State private var path: [AppRoute] = []

    var body: some View {
        NavigationStack(path: $path) {
            List {
                NavigationLink(value: AppRoute.home) {
                    Text("Home")
                }
                NavigationLink(value: AppRoute.settings) {
                    Text("Settings")
                }
            }
            .navigationDestination(for: AppRoute.self) { route in
                switch route {
                case .home:
                    HomeView(
                        model: HomeModel(loadHomeUseCase: container.loadHomeUseCase)
                    )
                case .settings:
                    SettingsView(model: SettingsModel())
                }
            }
        }
    }
}
```

## Integration Rules

- Create concrete data implementations in `AppContainer`.
- Inject only `Application` abstractions into `Presentation` models.
- Keep navigation in `App` (`AppRoute`, `RootView`), not in feature models.
- Keep `Presentation` free from direct `Data` imports.
- For bootstrap and previews, provide a lightweight placeholder use case (for example `PreviewLoadHomeUseCase`).
